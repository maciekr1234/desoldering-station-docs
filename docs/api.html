Enumerationsâ€‹

HeaterStateâ€‹

enum class HeaterState {
    Off,      // Heater disabled, PWM = 0
    Standby,  // PID control at standby_temperature from settings
    Sleep,    // Heater off, waiting for motion
    Running,  // Active PID control at user target
    Error,    // Fault detected, safe mode
    Tuning,   // Manual control for calibration
};

HeaterErrorâ€‹

enum class HeaterError {
    None = 0,           // No error
    NoHandle,           // Handle not connected (open circuit)
    SensorFailure,      // Temperature sensor reading invalid
    Overtemperature,    // Temperature exceeded maximum safe limit
    HeatingTimeout,     // Failed to reach target temperature in time
    RunawayHeating,     // Temperature rising uncontrollably
    UnderTemperature,   // Temperature dropped unexpectedly
};
