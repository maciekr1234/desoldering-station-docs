42static float read_heater_temperature_lut() {
43 float vref_mv = adc_input_get_millivolts(ADC_INPUT_VREF);
44 float millivolts = (adc_input_get_adc(ADC_INPUT_PTC_TEMP, true) * vref_mv) / 4095.0f;
45
46 // Check if handle is disconnected
47 if (millivolts >= CONFIG_PTC_NO_HANDLE_VOLTAGE_MV) {
48 return -1000.0f;
49 }
50
51 // Out of range checks - clamp to table boundaries
52 if (millivolts <= _heater_temp_lut[0].millivolts) {
53 return _heater_temp_lut[0].temperature;
54 }
55 if (millivolts >= _heater_temp_lut[_heater_temp_lut_count - 1].millivolts) {
56 return _heater_temp_lut[_heater_temp_lut_count - 1].temperature;
57 }
58
59 // Find the two entries to interpolate between
60 for (uint16_t i = 0; i < _heater_temp_lut_count - 1; i++) {
61 if (millivolts >= _heater_temp_lut[i].millivolts && millivolts <= _heater_temp_lut[i + 1].millivolts) {
62 // Linear interpolation between two points
63 float mv_diff = _heater_temp_lut[i + 1].millivolts - _heater_temp_lut[i].millivolts;
64 float temp_diff = _heater_temp_lut[i + 1].temperature - _heater_temp_lut[i].temperature;
65 float fraction = (millivolts - _heater_temp_lut[i].millivolts) / mv_diff;
66 return _heater_temp_lut[i].temperature + fraction * temp_diff;
67 }
68 }
69
70 // Should not reach here
71 return NAN;
72}

78static float read_ambient_temperature_ntc() {
79 float vref_adc = adc_input_get_adc(ADC_INPUT_VREF, true);
80 uint16_t adc = adc_input_get_adc(ADC_INPUT_AMBIENT_TEMP, true);
81
82 // Check for invalid readings to prevent division by zero or invalid calculations
83 if (adc >= 4095 || adc >= vref_adc) {
84 return 1000.0f;
85 }
86
87 // Calculate temperature using Beta equation for NTC thermistor
88 // T = 1 / (1/T0 + (1/B) * ln(R/R0)) - 273.15
89 float resistance = (_ntc_series_res * adc) / (vref_adc - adc);
90 float temp_k = 1.0f / ((log(resistance / _ntc_res25) / _ntc_beta) + (1.0f / (273.15f + 25.0f)));
91 return temp_k - 273.15f;
92}

102float read_heater_temperature_filtered() {
103 heater_settings_t* cfg = &gSettings.heater;
104 float temp_c = read_heater_temperature_celsius();
105
106 if (temp_c < 100.0f) {
107 temp_c = (temp_c) * (cfg->calibration.at100) / 100.0f;
108 } else if (temp_c < 200.0f) {
109 temp_c =
110 (temp_c - 100.0f) * (cfg->calibration.at200 - cfg->calibration.at100) / 100.0f + cfg->calibration.at100;
111 } else if (temp_c < 300.0f) {
112 temp_c =
113 (temp_c - 200.0f) * (cfg->calibration.at300 - cfg->calibration.at200) / 100.0f + cfg->calibration.at200;
114 } else if (temp_c < 350.0f) {
115 temp_c = (temp_c - 300.0f) * (cfg->calibration.at350 - cfg->calibration.at300) / 50.0f + cfg->calibration.at300;
116 } else if (temp_c < 400.0f) {
117 temp_c = (temp_c - 350.0f) * (cfg->calibration.at400 - cfg->calibration.at350) / 50.0f + cfg->calibration.at350;
118 } else if (temp_c < 450.0f) {
119 temp_c = (temp_c - 400.0f) * (cfg->calibration.at450 - cfg->calibration.at400) / 50.0f + cfg->calibration.at400;
120 }
121
122 return temp_c > 0.0f ? temp_c : 1000.f;
