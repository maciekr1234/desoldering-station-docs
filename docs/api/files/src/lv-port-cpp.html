91void lv_port_set_backlight(uint8_t pct) {
92 pct = constrain(pct, 0, 100);
93 lcd_bl_duty = pct;
94 uint16_t pwm = map(pct, 0, 100, 0, (1 << 12) - 1);
95 // Set PWM frequency to 20kHz for LCD backlight to avoid visible flicker
96
97 analogWrite(CONFIG_PIN_LCD_BL, pwm);
98}

106static void lcd_read_encoder(lv_indev_t* indev, lv_indev_data_t* data) {
107 data->state = LV_INDEV_STATE_RELEASED;
108 data->enc_diff = gEncoder.relative();
109
110 if (gEncoder.update_button()) {
111 data->state = LV_INDEV_STATE_PRESSED;
112 gHeater.triggerActivity();
113 }
114
115
116 if (data->enc_diff != 0) {
117 lv_display_trigger_activity(nullptr);
118 gHeater.triggerActivity();
119 if (_encoder_callback) {
120 _encoder_callback(data->enc_diff > 0 ? LV_PORT_INPUT_ROTATE_CW : LV_PORT_INPUT_ROTATE_CCW, data->enc_diff);
121 }
122 }

126static void lv_gui_task(void*) {
127 volatile uint32_t last_subject_update = 0;
128
129 for (;;) {
130 vTaskDelay(pdMS_TO_TICKS(5));
131
132 lv_timer_handler();
133
134 if( uint32_t now = millis(); now - last_subject_update >= 150) {
135 last_subject_update = now;
136 ui_update_global_subjects();
137 }
138
139 noInterrupts();
140 // Dynamically apply encoder inversion setting
141 gEncoder.set_inverted(gSettings.system.encoder_inverted);
142
143 // Adjust backlight based on heater state and settings
144 uint8_t brightness = gSettings.system.display_brightness;
145 switch (gHeater.getState()) {
146 case HeaterState::Sleep:
147 brightness = gSettings.system.sleep_brightness;
148 break;
149 case HeaterState::Standby:
150 brightness = gSettings.system.standby_brightness;
151 break;
152 case HeaterState::Running:
153 case HeaterState::Tuning:
154 case HeaterState::Error:
155 default:
156 brightness = gSettings.system.display_brightness;
157 break;
158 }
159 lv_port_set_backlight(brightness);
160 interrupts();
161 }
162}

164static void flush_cb(lv_display_t* disp, const lv_area_t* area, uint8_t* px_map) {
165 TFT_eSPI* tft = (TFT_eSPI*)lv_display_get_driver_data(disp);
166
167 uint32_t w = (area->x2 - area->x1 + 1);
168 uint32_t h = (area->y2 - area->y1 + 1);
169
170 tft->startWrite();
171 tft->setAddrWindow(area->x1, area->y1, w, h);
172 tft->pushColors((uint16_t*)px_map, w * h, true);
173 tft->endWrite();
174
175 lv_display_flush_ready(disp);
176}

179static void resolution_changed_event_cb(lv_event_t* e) {
180 lv_display_t* disp = (lv_display_t*)lv_event_get_target(e);
181 TFT_eSPI* tft = (TFT_eSPI*)lv_display_get_driver_data(disp);
182 int32_t hor_res = lv_display_get_horizontal_resolution(disp);
183 int32_t ver_res = lv_display_get_vertical_resolution(disp);
184 lv_display_rotation_t rot = lv_display_get_rotation(disp);
185
186 /* handle rotation */
187 switch (rot) {
188 case LV_DISPLAY_ROTATION_0:
189 tft->setRotation(0); /* Portrait orientation */
190 break;
191 case LV_DISPLAY_ROTATION_90:
192 tft->setRotation(1); /* Landscape orientation */
193 break;
194 case LV_DISPLAY_ROTATION_180:
195 tft->setRotation(2); /* Portrait orientation, flipped */
196 break;
197 case LV_DISPLAY_ROTATION_270:
198 tft->setRotation(3); /* Landscape orientation, flipped */
199 break;
200 }
201}
