323static uint8_t update_relay_high(float current_temp) {
324 g_ctx.last_temp = current_temp;
325 float target = static_cast<float>(g_ctx.config.target_temperature);
326
327 // Track maximum temperature in this half-cycle
328 if (current_temp > g_ctx.max_temp_in_half_cycle) {
329 g_ctx.max_temp_in_half_cycle = current_temp;
330 }
331
332 // Measure thermal lag: time from relay switch until temp starts rising significantly
333 if (g_ctx.thermal_lag_samples < 10 && g_ctx.temp_at_relay_switch > 0.0f) {
334 float temp_rise = current_temp - g_ctx.temp_at_relay_switch;
335 if (temp_rise > 1.0f) { // Temperature started responding
336 uint32_t lag = millis() - g_ctx.relay_switch_time_ms;
337 g_ctx.thermal_lag_sum_ms += lag;
338 g_ctx.thermal_lag_samples++;
339 g_ctx.temp_at_relay_switch = 0.0f; // Don't measure again this cycle
340 PICO_LOGD(TAG, "Thermal lag sample: %lu ms (rise=%.1fÂ°C)", lag, temp_rise);
341 }
342 }
343
344 // Switch to LOW when temperature exceeds target + hysteresis
345 if (current_temp >= target + g_ctx.config.hysteresis) {
346 // Record peak
347 record_point(g_ctx.max_temp_in_half_cycle, true);
348
349 // Record crossing time for period calculation
350 if (g_ctx.last_crossing_time_ms > 0) {
351 // Not the first crossing
352 }
353 g_ctx.last_crossing_time_ms = millis();
354
355 g_ctx.temp_at_relay_switch = current_temp;
356 enter_state(AUTO_TUNE_RELAY_LOW);
357 }
358
359 return g_ctx.current_power;
360}

365static uint8_t update_relay_low(float current_temp) {
366 g_ctx.last_temp = current_temp;
367 float target = static_cast<float>(g_ctx.config.target_temperature);
368
369 // Track minimum temperature in this half-cycle
370 if (current_temp < g_ctx.min_temp_in_half_cycle) {
371 g_ctx.min_temp_in_half_cycle = current_temp;
372 }
373
374 // Switch to HIGH when temperature drops below target - hysteresis
375 if (current_temp <= target - g_ctx.config.hysteresis) {
376 // Record trough
377 record_point(g_ctx.min_temp_in_half_cycle, false);
378
379 // Increment cycle count (one complete oscillation)
380 g_ctx.cycle_count++;
381
382 PICO_LOGI(TAG, "Cycle %d complete: peak=%.1fÂ°C, trough=%.1fÂ°C",
383 g_ctx.cycle_count,
384 g_ctx.max_temp_in_half_cycle,
385 g_ctx.min_temp_in_half_cycle);
386
387 // Check if we have enough cycles
388 if (g_ctx.cycle_count >= g_ctx.config.max_cycles) {
389 PICO_LOGI(TAG, "Maximum cycles reached, analyzing results");
390 enter_state(AUTO_TUNE_ANALYZING);
391 return 0;
392 }
393
394 if (g_ctx.cycle_count >= g_ctx.config.min_cycles) {
395 // Check if oscillation is stable (amplitude variance low)
396 float amplitude = calculate_average_amplitude();
397 if (amplitude > 0.0f && g_ctx.point_count >= 4) {
398 PICO_LOGI(TAG, "Sufficient stable cycles, analyzing results");
399 enter_state(AUTO_TUNE_ANALYZING);
400 return 0;
401 }
402 }
403
404 g_ctx.temp_at_relay_switch = current_temp;
405 enter_state(AUTO_TUNE_RELAY_HIGH);
406 }
407
408 return g_ctx.current_power;
409}

432static float calculate_average_period(void) {
433 if (g_ctx.point_count < 4) {
434 return 0.0f;
435 }
436
437 // Calculate period from peak-to-peak or trough-to-trough times
438 float period_sum = 0.0f;
439 uint8_t period_count = 0;
440
441 for (uint8_t i = 2; i < g_ctx.point_count; i++) {
442 // Find same type points (peak-to-peak or trough-to-trough)
443 if (g_ctx.points[i].is_peak == g_ctx.points[i-2].is_peak) {
444 uint32_t dt = g_ctx.points[i].timestamp_ms - g_ctx.points[i-2].timestamp_ms;
445 period_sum += static_cast<float>(dt) / 1000.0f; // Convert to seconds
446 period_count++;
447 }
448 }
449
450 if (period_count == 0) {
451 return 0.0f;
452 }
453
454 return period_sum / static_cast<float>(period_count);
455}

460static float calculate_average_amplitude(void) {
461 if (g_ctx.point_count < 2) {
462 return 0.0f;
463 }
464
465 float amplitude_sum = 0.0f;
466 uint8_t amplitude_count = 0;
467
468 // Calculate amplitude from consecutive peak-trough pairs
469 for (uint8_t i = 1; i < g_ctx.point_count; i++) {
470 if (g_ctx.points[i].is_peak != g_ctx.points[i-1].is_peak) {
471 float diff = fabsf(g_ctx.points[i].temperature - g_ctx.points[i-1].temperature);
472 amplitude_sum += diff;
473 amplitude_count++;
474 }
475 }
476
477 if (amplitude_count == 0) {
478 return 0.0f;
479 }
480
481 return amplitude_sum / static_cast<float>(amplitude_count);
482}

487static void analyze_results(void) {
488 PICO_LOGI(TAG, "Analyzing %d oscillation points from %d cycles",
489 g_ctx.point_count, g_ctx.cycle_count);
490
491 // Calculate ultimate period (Tu)
492 float Tu = calculate_average_period();
493 if (Tu <= 0.0f) {
494 PICO_LOGE(TAG, "Failed to calculate oscillation period");
495 enter_state(AUTO_TUNE_FAILED);
496 return;
497 }
498
499 // Calculate amplitude
500 float amplitude = calculate_average_amplitude();
501 if (amplitude <= 0.0f) {
502 PICO_LOGE(TAG, "Failed to calculate oscillation amplitude");
503 enter_state(AUTO_TUNE_FAILED);
504 return;
505 }
506
507 // Calculate relay amplitude (d) - this is half the power swing
508 float d = static_cast<float>(g_ctx.config.relay_power_high - g_ctx.config.relay_power_low) / 2.0f;
509
510 // Calculate ultimate gain (Ku) using relay method formula
511 // Ku = (4 * d) / (pi * amplitude)
512 float Ku = (4.0f * d) / (M_PI * amplitude);
513
514 // Calculate thermal lag
515 uint16_t thermal_lag_ms = 0;
516 if (g_ctx.thermal_lag_samples > 0) {
517 thermal_lag_ms = static_cast<uint16_t>(g_ctx.thermal_lag_sum_ms / g_ctx.thermal_lag_samples);
518 }
519
520 // Store results
521 g_ctx.result.ultimate_gain = Ku;
522 g_ctx.result.ultimate_period = Tu;
523 g_ctx.result.thermal_lag_ms = thermal_lag_ms;
524 g_ctx.result.amplitude = amplitude;
525 g_ctx.result.cycles_completed = g_ctx.cycle_count;
526
527 // Calculate PID parameters
528 calculate_pid_parameters();
529
530 PICO_LOGI(TAG, "Auto-tune complete: Ku=%.3f, Tu=%.2fs, amplitude=%.1fÂ°C, lag=%dms",
531 Ku, Tu, amplitude, thermal_lag_ms);
532 PICO_LOGI(TAG, "Ziegler-Nichols: Kp=%.3f, Ki=%.3f, Kd=%.3f",
533 g_ctx.result.pid_zn.kp, g_ctx.result.pid_zn.ki, g_ctx.result.pid_zn.kd);
534 PICO_LOGI(TAG, "Conservative: Kp=%.3f, Ki=%.3f, Kd=%.3f",
535 g_ctx.result.pid_conservative.kp, g_ctx.result.pid_conservative.ki,
536 g_ctx.result.pid_conservative.kd);
537
538 enter_state(AUTO_TUNE_COMPLETE);
539}
540
544static void calculate_pid_parameters(void) {
545 float Ku = g_ctx.result.ultimate_gain;
546 float Tu = g_ctx.result.ultimate_period;
547
548 // Ziegler-Nichols PID formula (classic)
549 // Kp = 0.6 * Ku
550 // Ki = 1.2 * Ku / Tu (or Ti = Tu/2, Ki = Kp/Ti)
551 // Kd = 0.075 * Ku * Tu (or Td = Tu/8, Kd = Kp*Td)
552 g_ctx.result.pid_zn.kp = 0.6f * Ku;
553 g_ctx.result.pid_zn.ki = 1.2f * Ku / Tu;
554 g_ctx.result.pid_zn.kd = 0.075f * Ku * Tu;
555 g_ctx.result.pid_zn.imax = 50.0f; // Reasonable defaults
556 g_ctx.result.pid_zn.imin = -5.0f;
557
558 // Conservative/No Overshoot PID formula
559 // More conservative tuning to reduce overshoot
560 // Kp = 0.33 * Ku
561 // Ki = 0.66 * Ku / Tu
562 // Kd = 0.11 * Ku * Tu
563 g_ctx.result.pid_conservative.kp = 0.33f * Ku;
564 g_ctx.result.pid_conservative.ki = 0.66f * Ku / Tu;
565 g_ctx.result.pid_conservative.kd = 0.11f * Ku * Tu;
566 g_ctx.result.pid_conservative.imax = 50.0f;
567 g_ctx.result.pid_conservative.imin = -5.0f;
568}
