97char* log_system_timestamp(void) {
98 static char buffer[18] = {0};
99 static SemaphoreHandle_t bufferLock = NULL;
100
101 if (bufferLock == NULL) {
102 bufferLock = xSemaphoreCreateMutex();

104
105 TickType_t ticks = xTaskGetTickCount();
106 uint32_t total_msec = (ticks * 1000) / configTICK_RATE_HZ;
107 uint32_t hours = (total_msec / (1000 * 60 * 60)) % 24;
108 uint32_t minutes = (total_msec / (1000 * 60)) % 60;
109 uint32_t seconds = (total_msec / 1000) % 60;
110 uint32_t msec = total_msec % 1000;
111
112 xSemaphoreTake(bufferLock, portMAX_DELAY);
113 snprintf(buffer, sizeof(buffer), "%02d:%02d:%02d.%03d", hours, minutes, seconds, msec);
