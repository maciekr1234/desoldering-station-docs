98void init_settings() {
99 if (!settings_mutex) {
100 settings_mutex = xSemaphoreCreateMutex();
101
102 if (!settings_mutex) {
103 PICO_LOGE(TAG, "Failed to create settings mutex");
104 return;
105 }
106 }
107
108 PICO_LOGD(TAG, "Initializing settings");
109
110 // Load settings from file
111 if (!load_global_settings()) {
112 PICO_LOGW(TAG, "Using default settings");
113 xSemaphoreTake(settings_mutex, portMAX_DELAY);
114 gSettings = default_settings;
115 xSemaphoreGive(settings_mutex);
116
117 save_global_settings();
118 }
119 settings_loaded = true;
120}

140bool load_settings(settings_t* settings, const char* filename) {
141 PICO_LOGI(TAG, "Loading settings from %s", filename);
142 if (!LittleFS.begin()) {
143 PICO_LOGE(TAG, "An Error has occurred while mounting LittleFS");
144 return false;
145 }
146
147 File file = LittleFS.open(filename, "r");
148 if (!file) {
149 PICO_LOGE(TAG, "Failed to open settings file for reading");
150 return false;
151 }
152
153 xSemaphoreTake(settings_mutex, portMAX_DELAY);
154
155 JsonDocument doc;
156 DeserializationError error = deserializeJson(doc, file);
157 file.close();
158 xSemaphoreGive(settings_mutex);
159 if (error) {
160 PICO_LOGE(TAG, "Failed to read settings file: %s", error.c_str());
161 return false;
162 }
163
164 xSemaphoreTake(settings_mutex, portMAX_DELAY);
165 settingsFromJSON(doc, *settings);
166 xSemaphoreGive(settings_mutex);
167
168 PICO_LOGI(TAG, "Settings loaded - calibration at100=%d at200=%d at300=%d",
169 settings->heater.calibration.at100,
170 settings->heater.calibration.at200,
171 settings->heater.calibration.at300);
172
173 return true;
174}

176bool save_settings(const settings_t* settings, const char* filename) {
177 PICO_LOGD(TAG, "Saving settings to %s", filename);
178 if (!LittleFS.begin()) {
179 PICO_LOGE(TAG, "An Error has occurred while mounting LittleFS");
180 return false;
181 }
182
183 File file = LittleFS.open(filename, "w");
184 if (!file) {
185 PICO_LOGE(TAG, "Failed to open settings file for writing");
186 return false;
187 }
188
189 xSemaphoreTake(settings_mutex, portMAX_DELAY);
190 JsonDocument doc = settingsToJSON(*settings);
191
192 if (serializeJsonPretty(doc, file) == 0) {
193 PICO_LOGE(TAG, "Failed to write settings to file");
194 xSemaphoreGive(settings_mutex);
195 file.close();
196 return false;
197 }
198 xSemaphoreGive(settings_mutex);
199
200 file.close();
201 return true;
202}

340static void settingsFromJSON(JsonDocument doc, settings_t& settings) {
341 uint8_t file_version = doc["version"] | 0;
342
343 // Version migration: if file version doesn't match, use defaults
344 if (file_version != SETTINGS_VERSION) {
345 PICO_LOGW(TAG, "Settings version mismatch (file=%d, current=%d), applying defaults",
346 file_version, SETTINGS_VERSION);
347 settings = default_settings;
348 return;
349 }
350
351 settings.version = SETTINGS_VERSION;
352
353 JsonObject heater = doc["heater"].as<JsonObject>();
354 heaterSettingsFromJSON(heater, settings.heater);
355
356 JsonObject pump = doc["pump"].as<JsonObject>();
357 pumpSettingsFromJSON(pump, settings.pump);
358
359 JsonObject system = doc["system"].as<JsonObject>();
360 systemSettingsFromJSON(system, settings.system);
361}
