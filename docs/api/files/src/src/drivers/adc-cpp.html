_adc_input_timer_callback()â€‹

static bool _adc_input_timer_callback (repeating_timer_t * rt)
static

Timer callback for periodic ADC processing.

This function reads the ADC values for each configured input, updates the circular buffer, calculates the average (with optional outlier rejection), and converts the average to millivolts.

Parameters
rtPointer to repeating_timer_t structure.
Returns

Always returns true to keep the timer running.

96static bool _adc_input_timer_callback(repeating_timer_t* rt) {
97 for (int i = 0; i < ADC_INPUT_COUNT; i++) {
98 int raw = analogRead(__FIRSTANALOGGPIO + i);
99 if (raw >= 0) {
100 adc_input_data_t* d = &adcInputData[i];
101 d->adc_raw = static_cast<uint16_t>(raw);
102 d->buf_sum -= d->buffer[d->buf_idx];
103 d->buffer[d->buf_idx] = d->adc_raw;
104 d->buf_sum += d->adc_raw;
105 d->buf_idx = (d->buf_idx + 1) % CONFIG_ADC_BUFFER_SIZE;
106 uint16_t min = d->buffer[0], max = d->buffer[0];
107 uint32_t sum = 0;
108 for (int j = 0; j < CONFIG_ADC_BUFFER_SIZE; j++) {
109 uint16_t val = d->buffer[j];
110 if (val < min)
111 min = val;
112 if (val > max)
113 max = val;
114 sum += val;
115 }
116#if CONFIG_ADC_OUTLIER_REJECTION
117 sum -= min;
118 sum -= max;
119 d->avg_adc = static_cast<uint16_t>(sum / (CONFIG_ADC_BUFFER_SIZE - 2));
120#else
121 d->avg_adc = static_cast<uint16_t>(d->buf_sum / CONFIG_ADC_BUFFER_SIZE);
122#endif
123 d->millivolts = (static_cast<uint32_t>(d->avg_adc) * 3300) / 4095;
124 }
125 }
126 return true;
127}
