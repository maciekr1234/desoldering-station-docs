96bool Encoder::update_button() {
97 uint32_t now_ms = millis();
98 bool now_button = gpio_get(_pin_btn);
99
100 _button_changed = false;
101
102 if (_btn_state == BTN_STATE_START && !now_button) {
103 _btn_state = BTN_STATE_DEBOUNCE;
104 _debounce_ms = now_ms;
105 }
106
107 if (_btn_state == BTN_STATE_DEBOUNCE) {
108 if (!now_button) {
109 if ((now_ms - _debounce_ms) > BUTTON_DEBOUNCE_MS) {
110 _btn_state = BTN_STATE_PRESSED;
111 _button_pressed = true;
112 _button_changed = true;
113 }
114 } else {
115 _button_pressed = false;
116 _btn_state = BTN_STATE_START;

118 }
119
120 if (_btn_state == BTN_STATE_PRESSED && now_button) {

127 _button_pressed = false;
