30bool _accelerationTimerCallback(repeating_timer_t* rt) {
31 MotorDriver* driver = static_cast<MotorDriver*>(rt->user_data);
32 auto* accel = &driver->_accel;

34
35 if (driver->_checkCommandTimeout()) {
36 // PICO_LOGD(TAG, "Command timeout occurred, motor disabled");
37 return true; // Continue the timer
38 }
39
40 if (!accel->ramping) {
41 return true; // Continue the timer
42 }
43
44 float t = static_cast<float>(accel->ramp_step) / static_cast<float>(accel->ramp_steps);
45 float progress = 0.5f * (1.0f - cosf(t * 3.14159265f)); // Cosine easing
46 float new_speed = accel->start_speed + (accel->target_speed - accel->start_speed) * progress;
47 float duty_cycle = fabsf(new_speed);
48
49 MotorDriver::State state = MotorDriver::State::Coast;

52 state = (new_speed >= 0.0f) ? MotorDriver::State::Forward : MotorDriver::State::Reverse;
53 }
54
55 driver->_updatePwm(duty_cycle, state);
56 accel->current_speed = new_speed;
57
58
59 accel->ramp_step++;
60 if (accel->ramp_step >= accel->ramp_steps) {
61 accel->current_speed = accel->target_speed;

73void MotorDriver::_updatePwm(float dutyCycle, State state) {
74 switch (state) {
75 case State::Forward: // Forward OUT1 high, OUT2 low
76 hw_pwm_set_duty(&_pwm1, dutyCycle);

102bool MotorDriver::_checkCommandTimeout() {
