31static float pid_calc_positional(pid_ctrl_block_t* pid, float error) {
32 float output = 0;
33 /* Add current error to the integral error */
34 pid->integral_err += error;
35 /* If the integral error is out of the range, it will be limited */
36 pid->integral_err = MIN(pid->integral_err, pid->max_integral);
37 pid->integral_err = MAX(pid->integral_err, pid->min_integral);
38
39 /* Calculate the pid control value by location formula */
40 /* u(k) = e(k)*Kp + (e(k)-e(k-1))*Kd + integral*Ki */
41 output = error * pid->Kp + (error - pid->previous_err1) * pid->Kd + pid->integral_err * pid->Ki;
42
43 /* If the output is out of the range, it will be limited */
44 output = MIN(output, pid->max_output);
45 output = MAX(output, pid->min_output);
46
47 /* Update previous error */
48 pid->previous_err1 = error;
49
50 return output;
51}

53static float pid_calc_incremental(pid_ctrl_block_t* pid, float error) {
54 float output = 0;
55
56 /* Calculate the pid control value by increment formula */
57 /* du(k) = (e(k)-e(k-1))*Kp + (e(k)-2*e(k-1)+e(k-2))*Kd + e(k)*Ki */
58 /* u(k) = du(k) + u(k-1) */
59 output = (error - pid->previous_err1) * pid->Kp + (error - 2 * pid->previous_err1 + pid->previous_err2) * pid->Kd +
60 error * pid->Ki + pid->last_output;
61
62 /* If the output is beyond the range, it will be limited */
63 output = MIN(output, pid->max_output);
64 output = MAX(output, pid->min_output);
65
66 /* Update previous error */
67 pid->previous_err2 = pid->previous_err1;
68 pid->previous_err1 = error;
69
70 /* Update last output */
71 pid->last_output = output;
72
73 return output;
74}

121pico_err_t pid_update_parameters(pid_ctrl_block_handle_t handle, const pid_ctrl_parameter_t* params) {
122 PICO_RETURN_ON_FALSE(handle && params, PICO_ERROR_INVALID_ARG, TAG, "invalid argument");
123 handle->Kp = params->kp;

128 handle->max_integral = params->max_integral;
129 handle->min_integral = params->min_integral;
130 /* Set the calculate function according to the PID type */
131 switch (params->cal_type) {
132 case PID_CAL_TYPE_INCREMENTAL:
