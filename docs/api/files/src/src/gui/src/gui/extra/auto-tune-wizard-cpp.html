lv_obj_t * create_auto_tune_subpage (lv_obj_t * menu)

Create auto-tune wizard subpage.

Creates a settings subpage with the auto-tune wizard UI. The wizard has three phases:

  • Configuration: Set target temperature and hysteresis
  • Running: Shows progress, cycles, ETA
  • Results: Displays calculated PID values, allows save/discard
Parameters
menuParent menu object
Returns

Pointer to the created page object

287static void tune_timer_cb(lv_timer_t* timer) {
288 if (!g_tune_ctx || g_tune_ctx->state != WizardState::Running) {
289 return;
290 }
291
292 auto_tune_state_t state = auto_tune_get_state();
293
294 // Check if tuning finished
295 if (state == AUTO_TUNE_COMPLETE || state == AUTO_TUNE_FAILED || state == AUTO_TUNE_CANCELLED ||
296 state == AUTO_TUNE_IDLE) {
297 bool success = (state == AUTO_TUNE_COMPLETE);
298
299 if (g_tune_ctx->timer) {
300 lv_timer_delete(g_tune_ctx->timer);
301 g_tune_ctx->timer = nullptr;
302 }
303
304 tune_build_complete_ui(success);
305 return;
306 }
307
308 // Update progress display
309 auto_tune_progress_t progress;
310 auto_tune_get_progress(&progress);
311 float current_temp = read_heater_temperature_filtered();
312
313 // Determine status text based on state
314 const char* status_text = "Running...";
315 switch (progress.state) {
316 case AUTO_TUNE_HEATING:
317 status_text = "Heating to target...";
318 break;
319 case AUTO_TUNE_RELAY_HIGH:
320 status_text = "Relay HIGH (heating)";
321 break;
322 case AUTO_TUNE_RELAY_LOW:
323 status_text = "Relay LOW (cooling)";
324 break;
325 case AUTO_TUNE_ANALYZING:
326 status_text = "Analyzing data...";
327 break;
328 default:
329 break;
330 }
331
332 // Update status with multi-line description
333 if (progress.eta_ms > 0) {
334 uint32_t remaining_sec = progress.eta_ms / 1000;
335 settings_item_set_description_fmt(g_tune_ctx->status_label,
336 "%s\n"
337 "Temp: %.1fÂ°C / %dÂ°C\n"
338 "Power: %d%% | Cycles: %d/10\n"
339 "ETA: %lum %lus",
340 status_text, current_temp, g_tune_ctx->target_temperature, progress.power_pct,
341 progress.cycle_count, remaining_sec / 60, remaining_sec % 60);
342 } else {
343 settings_item_set_description_fmt(g_tune_ctx->status_label,
344 "%s\n"
345 "Temp: %.1fÂ°C / %dÂ°C\n"
346 "Power: %d%% | Cycles: %d/10\n"
347 "ETA: Calculating...",
348 status_text, current_temp, g_tune_ctx->target_temperature, progress.power_pct,
349 progress.cycle_count);
350 }
351}

356static void tune_start_cb(lv_event_t* e) {

363
364 PICO_LOGI(TAG, "Starting auto-tune: target=%dÂ°C hysteresis=%.1fÂ°C", g_tune_ctx->target_temperature,

366
367 // Create auto-tune configuration
368 auto_tune_config_t config = auto_tune_default_config();
369 config.target_temperature = g_tune_ctx->target_temperature;

407static void tune_save_cb(lv_event_t* e) {
408 auto_tune_result_t result;
409 if (!auto_tune_get_result(&result)) {
410 create_msgbox_error("Error", "Failed to retrieve tuning results.");
411 return;
412 }
413
414 // Apply selected PID values to settings
415 const auto& pid = (g_tune_ctx && g_tune_ctx->selected_result == PidResultType::Conservative)
416 ? result.pid_conservative
417 : result.pid_zn;
418
419 gSettings.heater.pid.kp = pid.kp;
420 gSettings.heater.pid.ki = pid.ki;
421 gSettings.heater.pid.kd = pid.kd;
422
423 // Apply thermal lag
424 gSettings.heater.thermal_lag_ms = result.thermal_lag_ms;
425
426 // Save to flash
427 save_global_settings();
428
429 // Update heater PID
430 gHeater.setPID(pid.kp, pid.ki, pid.kd, gSettings.heater.pid.imax, gSettings.heater.pid.imin);
431
432 const char* type_name =
433 (g_tune_ctx && g_tune_ctx->selected_result == PidResultType::Conservative) ? "Conservative" : "Ziegler-Nichols";
434
435 PICO_LOGI(TAG, "Saved %s PID: Kp=%.3f Ki=%.3f Kd=%.3f lag=%dms", type_name, pid.kp, pid.ki, pid.kd,
436 result.thermal_lag_ms);
437
438 gHeater.changeState(HeaterState::Sleep);
439 create_msgbox_info("Success", "PID parameters saved!");
440
441 // Exit to main screen
442 lv_screen_pop();
443}
