343 lv_settings_item_t* item = &numeric->base;
344 if (!item->value_label)
345 return;

347 char buf[32];
348 if (numeric->decimal_places == 0) {
349 // Integer mode

351 } else {

353 int32_t multiplier = get_decimal_multiplier(numeric->decimal_places);

360

362 lv_snprintf(buf, sizeof(buf), "%s%" PRId32 ".%0*" PRId32 "%s", sign, int_part, (int)numeric->decimal_places, frac_part,
363 numeric->unit_suffix);
364 }

713static void settings_item_event_handler(const lv_obj_class_t* class_p, lv_event_t* e) {
714 lv_result_t res = lv_obj_event_base(class_p, e);
715 if (res != LV_RESULT_OK) {
716 return;
717 }
718
719 lv_event_code_t code = lv_event_get_code(e);
720 lv_obj_t* obj = lv_event_get_target_obj(e);
721 lv_settings_item_t* item = (lv_settings_item_t*)obj;
722
723
724 if (code == LV_EVENT_CLICKED) {
725 if (item->type == LV_SETTINGS_ITEM_TYPE_SWITCH) {
726 lv_settings_switch_t* sw = (lv_settings_switch_t*)obj;
727 lv_settings_item_set_switch_value(obj, !sw->state);
728 }
729 } else if (code == LV_EVENT_RELEASED) {
730 if (lv_group_get_editing(lv_obj_get_group(obj))) {
731 lv_group_set_editing(lv_obj_get_group(obj), false);
732 }
733
734 } else if (code == LV_EVENT_KEY) {
735 uint32_t key = lv_event_get_key(e);
736
737 if (item->type == LV_SETTINGS_ITEM_TYPE_NUMERIC) {
738 lv_settings_numeric_t* numeric = (lv_settings_numeric_t*)obj;
739 int32_t value = numeric->value;
740
741 if (key == LV_KEY_RIGHT || key == LV_KEY_UP) {
742 value += numeric->step;
743 } else if (key == LV_KEY_LEFT || key == LV_KEY_DOWN) {
744 value -= numeric->step;
745 }
746 lv_settings_item_set_numeric_value(obj, value);
747 } else if (item->type == LV_SETTINGS_ITEM_TYPE_SELECT) {
748 lv_settings_select_t* select = (lv_settings_select_t*)obj;
749 if (select->options_count == 0)
750 return;
751
752 uint32_t new_index = select->current_index;
753 if (key == LV_KEY_RIGHT || key == LV_KEY_UP) {
754 // Wrap around forward
755 new_index = (new_index + 1) % select->options_count;
756 } else if (key == LV_KEY_LEFT || key == LV_KEY_DOWN) {
757 // Wrap around backward
758 if (new_index == 0) {
759 new_index = select->options_count - 1;
760 } else {
761 new_index--;
762 }
763 }
764 lv_settings_select_set_index(obj, new_index);
765 }
766 }

864void lv_settings_select_set_options(lv_obj_t* obj, const char* options) {
865 if (!obj || !lv_obj_has_class(obj, &lv_settings_select_class)) {
866 PICO_LOGE(TAG, "Object is not a select settings item");
867 return;
868 }
869
870 lv_settings_select_t* select = (lv_settings_select_t*)obj;
871 lv_settings_item_t* item = (lv_settings_item_t*)obj;
872
873 // Clear existing options
874 settings_select_clear_options(obj);
875
876 // Handle NULL or empty string
877 if (!options || options[0] == '\0') {
878 if (item->value_label) {
879 lv_label_set_text(item->value_label, "");
880 }
881 return;
882 }
883
884 // Count options (number of '\n' + 1)
885 uint32_t count = 1;
886 for (const char* p = options; *p; p++) {
887 if (*p == '\n')
888 count++;
889 }
890
891 // Allocate array of string pointers
892 select->options = (char**)lv_malloc(count * sizeof(char*));
893 if (!select->options) {
894 PICO_LOGE(TAG, "Failed to allocate options array");
895 return;
896 }
897
898 // Parse and copy each option
899 const char* start = options;
900 uint32_t idx = 0;
901 while (*start && idx < count) {
902 // Find end of current option
903 const char* end = start;
904 while (*end && *end != '\n')
905 end++;
906
907 // Calculate length and allocate
908 size_t len = end - start;
909 select->options[idx] = (char*)lv_malloc(len + 1);
910 if (!select->options[idx]) {
911 PICO_LOGE(TAG, "Failed to allocate option string");
912 // Clean up already allocated strings
913 for (uint32_t i = 0; i < idx; i++) {
914 lv_free(select->options[i]);
915 }
916 lv_free(select->options);
917 select->options = NULL;
918 return;
919 }
920
921 // Copy the string
922 lv_memcpy(select->options[idx], start, len);
923 select->options[idx][len] = '\0';
924
925 idx++;
926 start = (*end == '\n') ? end + 1 : end;
927 }
928
929 select->options_count = idx;
930 select->options_static = false;
931 select->current_index = 0;
932
933 // Update display
934 if (item->value_label && select->options_count > 0) {
935 lv_label_set_text(item->value_label, select->options[0]);
936 }
937}
