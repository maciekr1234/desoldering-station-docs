229static void page_exit_cb(lv_event_t* e) {
230 lv_obj_t* btn = lv_event_get_target_obj(e);
231 // Navigate up parent hierarchy to find the menu
232 lv_obj_t* obj = btn;
233 lv_menu_t* menu = nullptr;
234 while (obj) {
235 if (lv_obj_check_type(obj, &lv_menu_class)) {
236 menu = (lv_menu_t*)obj;
237 break;
238 }
239 obj = lv_obj_get_parent(obj);
240 }
241
242 if (menu) {
243 // Check menu depth - cur_depth == 0 means we're at root
244 if (menu->cur_depth == 0) {
245 // At root - handle settings exit with dirty check
246 handle_settings_exit();
247 } else {
248 // Not at root - simulate back button press to go back one level
249 lv_obj_t* back_btn = lv_menu_get_main_header_back_button((lv_obj_t*)menu);
250 if (back_btn) {
251 lv_obj_send_event(back_btn, LV_EVENT_CLICKED, nullptr);
252
253 // Focus first item on the new page
254 lv_obj_t* cur_page = lv_menu_get_cur_main_page((lv_obj_t*)menu);
255 if (cur_page) {
256 // Page -> Section -> First item
257 lv_obj_t* section = lv_obj_get_child(cur_page, 0);
258 if (section) {
259 lv_obj_t* first_item = lv_obj_get_child(section, 0);
260 if (first_item) {
261 lv_group_focus_obj(first_item);
262 }
263 }
264 }
265 }
266 }
267 }

475static uint32_t get_frequency_divisor(FrequencyScale scale) {
476 switch (scale) {
477 case FrequencyScale::Hz:
478 return 1;
479 case FrequencyScale::kHz:
480 return 1000;
481 case FrequencyScale::MHz:
482 return 1000000;
483 case FrequencyScale::GHz:
484 return 1000000000;
485 default:
486 return 1;
487 }
488}

490static const char* get_frequency_unit(FrequencyScale scale) {
491 switch (scale) {
492 case FrequencyScale::Hz:
493 return " Hz";
494 case FrequencyScale::kHz:
495 return " kHz";
496 case FrequencyScale::MHz:
497 return " MHz";
498 case FrequencyScale::GHz:
499 return " GHz";
500 default:
501 return " Hz";
502 }
503}

509static FrequencyScale auto_select_frequency_scale(uint32_t max_hz) {
510 if (max_hz >= 1000000000)
511 return FrequencyScale::GHz;
512 if (max_hz >= 1000000)
513 return FrequencyScale::MHz;
514 if (max_hz >= 1000)
515 return FrequencyScale::kHz;
516 return FrequencyScale::Hz;
517}

519lv_obj_t* create_frequency_bound(lv_obj_t* parent,
520 const char* title,
521 uint32_t* value,
522 uint32_t min_hz,
523 uint32_t max_hz,
524 uint32_t step_hz,
525 FrequencyScale scale) {
526 // Auto-select scale if needed
527 if (scale == FrequencyScale::Auto) {
528 scale = auto_select_frequency_scale(max_hz);
529 }
530
531 uint32_t divisor = get_frequency_divisor(scale);
532
533 // Allocate bind data (will be freed when widget is deleted via destructor)
534 FrequencyBindData* data = static_cast<FrequencyBindData*>(lv_malloc(sizeof(FrequencyBindData)));
535 data->value_ptr = value;
536 data->divisor = divisor;
537
538 // Convert from Hz to display scale
539 int32_t display_value = static_cast<int32_t>(*value / divisor);
540 int32_t display_min = static_cast<int32_t>(min_hz / divisor);
541 int32_t display_max = static_cast<int32_t>(max_hz / divisor);
542 int32_t display_step = static_cast<int32_t>(step_hz / divisor);
543 if (display_step < 1)
544 display_step = 1;
545
546 lv_obj_t* obj = lv_settings_item_create_numeric(parent, title, display_value);
547 lv_settings_item_set_numeric_range(obj, display_min, display_max);
548 lv_settings_item_set_numeric_step(obj, display_step);
549 lv_settings_item_set_numeric_unit(obj, get_frequency_unit(scale));
550 lv_obj_add_event_cb(obj, frequency_bound_cb, LV_EVENT_VALUE_CHANGED, data);
551
552 // Free the data when the object is deleted
553 lv_obj_add_event_cb(
554 obj,
555 [](lv_event_t* e) {
556 FrequencyBindData* d = static_cast<FrequencyBindData*>(lv_event_get_user_data(e));
557 lv_free(d);
558 },
559 LV_EVENT_DELETE, data);
560
561 return obj;
562}
