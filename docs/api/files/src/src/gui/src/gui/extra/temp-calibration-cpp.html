303static void calibration_timer_cb(lv_timer_t* timer) {
304 if (g_calib_ctx.state == CalibrationState::Complete || g_calib_ctx.state == CalibrationState::Idle ||
305 g_calib_ctx.state == CalibrationState::Timeout) {
306 return;
307 }
308
309 int32_t current_temp = static_cast<int32_t>(read_heater_temperature_filtered());
310 uint16_t target = CALIBRATION_TARGETS[g_calib_ctx.step_index];
311 uint32_t now = millis();
312
313
314 // Update temperature display
315 calibration_wizard_ui_set_status(current_temp, nullptr);
316
317 // if (g_calib_ctx.temp_label) {
318
319 // settings_item_set_description_fmt(g_calib_ctx.temp_label, "%ld C", current_temp);
320 // }
321
322
323 // Check for timeout (5 minutes)
324 if (now - g_calib_ctx.step_start_ms > CALIBRATION_TIMEOUT_MS) {
325 PICO_LOGW(TAG, "Calibration timeout at step %d", g_calib_ctx.step_index + 1);
326 g_calib_ctx.state = CalibrationState::Timeout;
327
328 // Update status
329 if (g_calib_ctx.status_label) {
330 calibration_wizard_ui_set_status(0, "Timeout...");
331 // settings_item_set_description_fmt(g_calib_ctx.status_label, "Timeout...");
332 }
333
334 // Turn off heater
335 gHeater.changeState(HeaterState::Sleep);
336
337 // Delete this timer and schedule exit
338 lv_timer_delete(g_calib_ctx.timer);
339 g_calib_ctx.timer = NULL;
340 lv_timer_create(calibration_timeout_exit_cb, 2000, NULL);
341 return;
342 }
343
344 // Check temperature stability
345 int16_t diff = abs(current_temp - target);
346
347 if (diff <= STABILITY_THRESHOLD) {
348 // Temperature within range
349 if (g_calib_ctx.state == CalibrationState::Heating) {
350 g_calib_ctx.state = CalibrationState::Stabilizing;
351 g_calib_ctx.stability_start_ms = now;
352
353 if (g_calib_ctx.status_label) {
354 // settings_item_set_description_fmt(g_calib_ctx.status_label, "Stabilizing...");
355 calibration_wizard_ui_set_status(0, "Stabilizing...");
356 }
357 } else if (g_calib_ctx.state == CalibrationState::Stabilizing) {
358 // Check if stable for required duration
359 if (now - g_calib_ctx.stability_start_ms >= STABILITY_DURATION_MS) {
360 g_calib_ctx.state = CalibrationState::WaitingInput;
361
362 // Enable input and confirm button
363 if (g_calib_ctx.input) {
364 lv_obj_remove_state(g_calib_ctx.input, LV_STATE_DISABLED);
365 }
366 if (g_calib_ctx.confirm_btn) {
367 lv_obj_remove_state(g_calib_ctx.confirm_btn, LV_STATE_DISABLED);
368 }
369
370 if (g_calib_ctx.status_label) {
371 calibration_wizard_ui_set_status(0, "Ready");
372 // settings_item_set_description_fmt(g_calib_ctx.status_label, "Ready");
373 }
374
375 PICO_LOGI(TAG, "Step %d ready for input", g_calib_ctx.step_index + 1);
376 }
377 }
378 } else {
379 // Temperature outside range, reset stability
380 if (g_calib_ctx.state == CalibrationState::Stabilizing) {
381 g_calib_ctx.state = CalibrationState::Heating;
382 g_calib_ctx.stability_start_ms = 0;
383
384 if (g_calib_ctx.status_label) {
385 // settings_item_set_description_fmt(g_calib_ctx.status_label, "Heating...");
386 calibration_wizard_ui_set_status(0, "Heating...");
387 }
388 }
389 }

395static void calibration_confirm_cb(lv_event_t* e) {
396 if (g_calib_ctx.state != CalibrationState::WaitingInput) {
397 return;
398 }
399
400 // Store measured value
401 int32_t measured = lv_settings_item_get_value(g_calib_ctx.input);

403

406
407 // Advance to next step
408 g_calib_ctx.step_index++;
409

421 // All steps complete

423

427 g_calib_ctx.timer = NULL;
428 }
429
430 // Turn off heater
431 gHeater.changeState(HeaterState::Sleep);
432
433 // Show results
434 calibration_build_complete_ui();
435 }
