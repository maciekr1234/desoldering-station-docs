116static void pid_apply_cb(lv_event_t* e) {
117 float kp = lv_settings_item_get_value_float(pid_kp_input);
118 float ki = lv_settings_item_get_value_float(pid_ki_input);
119 float kd = lv_settings_item_get_value_float(pid_kd_input);
120 float imax = lv_settings_item_get_value_float(pid_imax_input);
121 float imin = lv_settings_item_get_value_float(pid_imin_input);
122
123 gHeater.setPID(kp, ki, kd, imax, imin);
124 PICO_LOGI(TAG, "PID applied: Kp=%.2f Ki=%.2f Kd=%.2f IMax=%.1f IMin=%.1f", kp, ki, kd, imax, imin);
125}
126
127static void pid_save_cb(lv_event_t* e) {
128 float kp = lv_settings_item_get_value_float(pid_kp_input);
129 float ki = lv_settings_item_get_value_float(pid_ki_input);
130 float kd = lv_settings_item_get_value_float(pid_kd_input);
131 float imax = lv_settings_item_get_value_float(pid_imax_input);
132 float imin = lv_settings_item_get_value_float(pid_imin_input);
133
134 // Apply to heater
135 gHeater.setPID(kp, ki, kd, imax, imin);
136
137 // Save to settings
138 gSettings.heater.pid.kp = kp;
139 gSettings.heater.pid.ki = ki;
140 gSettings.heater.pid.kd = kd;
141 gSettings.heater.pid.imax = imax;
142 gSettings.heater.pid.imin = imin;
143 save_global_settings();
144
145 // Show success toast
146 create_msgbox_info("Success", "PID settings saved!");
147
148 PICO_LOGI(TAG, "PID saved: Kp=%.2f Ki=%.2f Kd=%.2f IMax=%.1f IMin=%.1f", kp, ki, kd, imax, imin);
149}

151static lv_obj_t* create_pid_tuning_subpage(lv_obj_t* menu) {
152 auto [page, section] = create_page(menu, "PID Tuning");
153
154 // Get current PID parameters
155 pid_ctrl_parameter_t params = gHeater.getPIDParameters();
156
157 // Create float numeric inputs for each parameter
158 pid_kp_input = create_numeric_float(section, "Kp (Proportional)", params.kp, 0.0f, 100.0f, 2, 1);
159 pid_ki_input = create_numeric_float(section, "Ki (Integral)", params.ki, 0.0f, 10.0f, 2, 1);
160 pid_kd_input = create_numeric_float(section, "Kd (Derivative)", params.kd, 0.0f, 50.0f, 1, 1);
161 pid_imax_input = create_numeric_float(section, "I-Max", params.max_integral, 0.0f, 100.0f, 1, 1);
162 pid_imin_input = create_numeric_float(section, "I-Min", params.min_integral, -100.0f, 0.0f, 1, 1);
163
164 // Apply button (applies to heater but doesn't save)
165 create_button(section, "Apply", pid_apply_cb);
166
167 // Save button (applies and saves to flash)
168 create_button(section, "Save to Flash", pid_save_cb);
169
170 return page;
171}
