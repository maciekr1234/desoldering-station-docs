Configurable Timeouts

Heating, standby, and sleep timeouts are now configurable via the settings menu. The values are stored in gSettings.heater.heating_timeout_ms, gSettings.heater.standby_timeout_ms, and gSettings.heater.sleep_timeout_ms. Default values are 5, 10, and 15 minutes respectively.

Settings Structureâ€‹

Runtime configuration stored in gSettings:

include/settings.h
struct Settings {
    struct {
        uint16_t target_temperature;      // Default: 350Â°C
        uint8_t  standby_temperature;     // Default: 200Â°C
        uint16_t standby_timeout_sec;     // Default: 60s
        uint16_t sleep_timeout_sec;       // Default: 300s
        struct {
            float kp, ki, kd;             // PID gains
            float imax, imin;             // Integral limits
        } pid;
    } heater;
    
    struct {
        uint8_t  speed_percent;           // Default: 100%
        uint16_t acceleration_ms;         // Default: 500ms
    } pump;
    
    struct {
        uint8_t  brightness;              // Default: 80%
        bool     dark_mode;               // Default: true
    } display;
};

extern Settings gSettings;
extern SemaphoreHandle_t gSettingsMutex;

Build Configurationâ€‹

PlatformIO Environmentâ€‹

platformio.ini
[env:desoldering-station-rev3]
platform = https://github.com/maxgerhardt/platform-raspberrypi.git
board = rpipico
framework = arduino

; FreeRTOS SMP
board_build.core = earlephilhower
build_flags = 
    -DUSE_FREERTOS
    -DCONFIG_FREERTOS_UNICORE=0
    
; LittleFS partition
board_build.filesystem_size = 0.5m

; Upload settings
upload_protocol = picotool
monitor_speed = 115200

Build Commandsâ€‹

# Build only
pio run -e desoldering-station-rev3

# Build and upload
pio run -e desoldering-station-rev3 -t upload

# Upload filesystem
pio run -e desoldering-station-rev3 -t uploadfs

# Clean build
pio run -e desoldering-station-rev3 -t clean

# Debug build
pio debug -e desoldering-station-rev3
