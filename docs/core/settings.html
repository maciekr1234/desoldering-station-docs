Settings System

Persistent configuration storage using LittleFS.

Overviewâ€‹

The Settings system provides a centralized configuration store (gSettings) with JSON persistence to flash storage using LittleFS and ArduinoJson.

Featuresâ€‹

  • Global settings structure with defaults
  • JSON serialization/deserialization
  • LittleFS persistence
  • Versioned settings format
  • Reset to defaults support

Architectureâ€‹

Settings Structureâ€‹

include/core/settings.h
#define SETTINGS_VERSION 2

typedef enum {
    HEATER_STARTUP_SLEEP   = 0, // Heater stays in sleep mode at startup
    HEATER_STARTUP_STANDBY = 1, // Heater enters standby at startup
    HEATER_STARTUP_ON      = 2, // Heater starts heating immediately
} heater_startup_state_t;

typedef struct {
    uint32_t frequency_hz;     // Pump PWM frequency in Hz
    uint32_t acceleration_ms;  // Pump acceleration time in ms
} pump_settings_t;

typedef struct {
    uint16_t max_temperature;      // Maximum allowable temperature
    uint16_t min_temperature;      // Minimum allowable temperature
    uint16_t default_temperature;  // Default target temperature
    uint16_t standby_temperature;  // Standby temperature
    uint32_t heating_timeout_ms;   // Timeout to reach target (ms)
    uint32_t standby_timeout_ms;   // Inactivity timeout before standby (ms)
    uint32_t sleep_timeout_ms;     // Inactivity timeout before sleep (ms)
    uint8_t  startup_state;        // Heater state at startup (heater_startup_state_t)
    
    pid_tunings_t pid;  // PID controller parameters
    
    struct {  // Temperature calibration
        uint16_t at100, at200, at300, at350, at400, at450;
    } calibration;
} heater_settings_t;

typedef struct {
    bool    encoder_inverted;    // Invert encoder direction
    uint8_t display_brightness;  // Normal brightness (0-100)
    uint8_t standby_brightness;  // Standby brightness (0-100)
    uint8_t sleep_brightness;    // Sleep brightness (0-100)
} system_settings_t;

typedef struct _settings_t {
    uint8_t           version;  // Settings format version
    heater_settings_t heater;   // Heater-related settings
    pump_settings_t   pump;     // Pump motor settings
    system_settings_t system;   // System/UI settings
} settings_t;

extern settings_t gSettings;
extern const settings_t default_settings;

Example Usageâ€‹

Startup Initializationâ€‹

void setup() {
    // Initialize settings system
    init_settings();
    
    // Load from flash (uses defaults if not found)
    if (!load_global_settings()) {
        PICO_LOGW("SETTINGS", "Using defaults, saving...");
        save_global_settings();
    }
    
    // Apply loaded settings
    heater.setTargetTemperature(gSettings.heater.default_temperature);
}

Reading Settingsâ€‹

void apply_pid_settings() {
    pid_ctrl_parameter_t params = {
        .kp = gSettings.heater.pid.kp,
        .ki = gSettings.heater.pid.ki,
        .kd = gSettings.heater.pid.kd,
        .max_output = gSettings.heater.pid.max_output,
        .min_output = gSettings.heater.pid.min_output,
        .max_integral = gSettings.heater.pid.max_integral,
        .min_integral = gSettings.heater.pid.min_integral,
        .cal_type = PID_CAL_TYPE_POSITIONAL
    };
    heater.updatePID(params);
}

Modifying Settingsâ€‹

void set_temperature(uint16_t temp) {
    gSettings.heater.default_temperature = temp;
    save_global_settings();  // Persist to flash
}

Adding New Settingsâ€‹

  1. Add to structure in include/core/settings.h:

    typedef struct {
        // ... existing ...
        uint8_t new_setting;
    } heater_settings_t;

  2. Update default_settings in src/core/settings.cpp:

    const settings_t default_settings = {
        .heater = {
            // ... existing ...
            .new_setting = 42,
        },
    };

  3. Update serialization in src/core/settings.cpp:

    // In save function
    heater["new_setting"] = settings->heater.new_setting;

    // In load function
    settings->heater.new_setting = heater["new_setting"] | default_settings.heater.new_setting;
