Settings Menu Widget

Custom LVGL widget for hierarchical settings menus.

Overviewâ€‹

lv_settings_menu is a custom widget that provides a hierarchical settings interface with multiple item types, LVGL subject binding, and encoder-friendly navigation.

Featuresâ€‹

  • Hierarchical menu structure with pages and sections
  • Multiple item types (text, button, switch, numeric, select, subpage)
  • Automatic lv_subject binding for reactive updates
  • Customizable switch display modes
  • Numeric items with step, range, and format options

Item Typesâ€‹

TypeEnumDescription
TextLV_SETTINGS_ITEM_TYPE_TEXTStatic text with description
ButtonLV_SETTINGS_ITEM_TYPE_BUTTONClickable action button
SubpageLV_SETTINGS_ITEM_TYPE_SUBPAGEOpens another settings page

Subject Bindingâ€‹

lv_observer_t* lv_settings_select_bind_index(lv_obj_t* obj, 
                                              lv_subject_t* subject);

// Bind select text to string subject (one-way: widget -> subject)
lv_observer_t* lv_settings_select_bind_value(lv_obj_t* obj, 
                                              lv_subject_t* subject);

Example Usageâ€‹

Basic Menu with Pagesâ€‹

// Create main menu
lv_obj_t* menu = lv_settings_menu_create(lv_scr_act());

// Create main page
lv_obj_t* main_page = lv_settings_page_create(menu, "Settings");

// Add sections and items
lv_obj_t* section = lv_settings_section_create(main_page);

// Switch item
lv_obj_t* dark_mode = lv_settings_item_create_switch(section, "Dark Mode", true);
lv_settings_item_set_switch_mode(dark_mode, LV_SETTINGS_ITEM_SWITCH_ON_OFF);

// Integer numeric item with unit suffix
lv_obj_t* temp = lv_settings_item_create_numeric(section, "Temperature", 350);
lv_settings_item_set_numeric_range(temp, 100, 450);
lv_settings_item_set_numeric_step(temp, 5);
lv_settings_item_set_numeric_unit(temp, " Â°C");

Decimal / Floating-Point Valuesâ€‹

The numeric widget supports fixed-point decimal values using decimal_places. Internally, values are stored as integers for precision, but displayed with a decimal point.

// PID gain with 2 decimal places (displayed as "12.50")
lv_obj_t* kp = lv_settings_item_create_numeric(section, "Kp", 1250);
lv_settings_item_set_numeric_decimal_places(kp, 2);  // 1250 -> "12.50"
lv_settings_item_set_numeric_range(kp, 0, 10000);    // 0.00 to 100.00
lv_settings_item_set_numeric_step(kp, 1);            // Step by 0.01

// Using float convenience functions (auto-converts to internal representation)
lv_settings_item_set_numeric_decimal_places(kp, 2);
lv_settings_item_set_numeric_range_float(kp, 0.0f, 100.0f);
lv_settings_item_set_numeric_value_float(kp, 12.5f);
float current_kp = lv_settings_item_get_value_float(kp);  // Returns 12.5f

Float Subject Bindingâ€‹

// Create a float subject
static lv_subject_t subject_kp;
lv_subject_init_float(&subject_kp, 12.5f);

// Create numeric item with 2 decimal places
lv_obj_t* kp = lv_settings_item_create_numeric(section, "Kp", 0);
lv_settings_item_set_numeric_decimal_places(kp, 2);
lv_settings_item_set_numeric_range_float(kp, 0.0f, 100.0f);

// Bind to float subject - auto-converts between internal int and float
lv_settings_numeric_bind_value(kp, &subject_kp);
// Widget shows "12.50", subject contains 12.5f

Select Item with Optionsâ€‹

lv_obj_t* unit_select = lv_settings_select_create(section, "Temperature Unit");

// Option 1: Newline-separated string (copied)
lv_settings_select_set_options(unit_select, "Celsius\nFahrenheit\nKelvin");

// Option 2: Static array (no copy, must remain valid)
static const char* units[] = {"Celsius", "Fahrenheit", "Kelvin"};
lv_settings_select_set_options_static(unit_select, units, 3);

lv_settings_select_set_index(unit_select, 0);  // Select "Celsius"

Subpage Navigationâ€‹

// Callback to create PID settings page
lv_obj_t* create_pid_page(lv_obj_t* menu) {
    lv_obj_t* page = lv_settings_page_create(menu, "PID Settings");
    lv_obj_t* section = lv_settings_section_create(page);
    
    // PID gains with 2 decimal places
    lv_obj_t* kp = lv_settings_item_create_numeric(section, "Kp", 1000);
    lv_settings_item_set_numeric_decimal_places(kp, 2);
    lv_settings_item_set_numeric_range_float(kp, 0.0f, 100.0f);
    
    lv_obj_t* ki = lv_settings_item_create_numeric(section, "Ki", 100);
    lv_settings_item_set_numeric_decimal_places(ki, 2);
    lv_settings_item_set_numeric_range_float(ki, 0.0f, 100.0f);
    
    lv_obj_t* kd = lv_settings_item_create_numeric(section, "Kd", 200);
    lv_settings_item_set_numeric_decimal_places(kd, 2);
    lv_settings_item_set_numeric_range_float(kd, 0.0f, 100.0f);
    
    return page;
}

// Add subpage item
lv_settings_item_create_subpage(section, "PID Settings", create_pid_page);

Data Binding with Subjectsâ€‹

// Create subjects
static lv_subject_t subject_brightness;
static lv_subject_t subject_encoder_inverted;

lv_subject_init_int(&subject_brightness, 100);
lv_subject_init_int(&subject_encoder_inverted, 0);

// Create items and bind
lv_obj_t* brightness = lv_settings_item_create_numeric(section, "Brightness", 100);
lv_settings_item_set_numeric_range(brightness, 0, 100);
lv_settings_numeric_bind_value(brightness, &subject_brightness);

lv_obj_t* encoder = lv_settings_item_create_switch(section, "Invert Encoder", false);
lv_settings_switch_bind_value(encoder, &subject_encoder_inverted);

// Widget changes automatically update the subject
